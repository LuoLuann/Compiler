function whileTest(n: integer) {
    while(n != 0) {
        print(n);
        n = n - 1;
    }
}

function procedureTest() {
    print(10);
}

function funcaoTest(): integer {
    print(10);
    return 2;
}

function forTest() {
    for(var i: integer = 0; i != 10; i = i + 1) {
        print(i + 2);
    }
}

function constTest() {
    const PI: integer = 3;
    print(PI);
}

function conditionalTest(x: integer, y: integer) {
    if (x > y) {
        print(x);
    } else if (x == y) {
        print(0);
    } else {
        print(y);
    }
}

function sum(a: integer, b: integer): integer {
    return a + b;
}

function callFunctionTest() {
    var result: integer = sum(10, 20);
    print(result);
}

function loopControlTest() {
    for (var i: integer = 0; i < 10; i = i + 1) {
        if (i == 5) {
            break;
        }
        if (i / 2 == 0) {
            continue;
        }
        print(i);
    }

    var j: integer = 0;
    
    while (j < 5) {
        j = j + 1;
        if (j == 3) {
            continue;
        }
        print(j);
    }
}

function nestedLoopsTest() {
    for (var i: integer = 0; i < 3; i = i + 1) {
        while (i < 2) {
            print(i);
            break;
        }
    }
}

    return width * height;
}

function areaTest() {
    var w: integer = 5;
    var h: integer = 10;
    var area: integer = calculateArea(w + 2, h - 3);
    print(area);
}

}
